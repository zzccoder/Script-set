--================================================================

--查看表的锁升级设置

SELECT name AS TableName,

lock_escalation_desc

FROM sys.tables

WHERE name='tableName'


--================================================================

--修改表的锁升级设置

ALTER TABLE TableName 

SET (LOCK_ESCALATION = TABLE | AUTO | DISABLE);


--===============================================================

--在表分区后，建议将表的锁升级选项设置为AUTO

ALTER TABLE TableName 

SET LOCK_ESCALATION=AUTO

--在SQL 2014版本上，需要写成

ALTER TABLE [TableName] 

SET (LOCK_ESCALATION=AUTO)
