--查看指定数据库当前最小 LSN


DECLARE @database_name NVARCHAR( 200)

SET @database_name ='DBName'

SELECT  MIN (redo_start_lsn) 

FROM master .sys. master_files 

WHERE database_id = DB_ID ( @database_name)

AND TYPE = 0

AND STATE IN (0, 1)

AND redo_start_lsn IS NOT NULL
