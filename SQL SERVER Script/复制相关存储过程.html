--返回发布服务器上属于一个或多个发布的订阅的当前状态信息，

--并为每个返回的订阅返回一行。 在分发服务器上对分发数据库

--执行此存储过程，用于监视复制。


--@publication_type=0:事务发布

--@mode=3:只返回带错误或已生成在达到阈值度量指标时发出的警告的订阅。

EXEC distribution.dbo.sp_replmonitorhelpsubscription  

@publisher = null,

@publisher_db = null,

@publication = null,

@publication_type =0,

@mode = 3,

@topnum = 0,

@exclude_anonymous = null,

@refreshpolicy = 0
