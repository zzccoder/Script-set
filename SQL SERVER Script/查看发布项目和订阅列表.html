SELECT  a.name AS [发布服务器名] ,

        b.publisher_db AS [发布数据库] ,

        b.publisher_id AS [发布编号] ,

        b.publication AS [发布名] ,

        c.source_object AS [项目名] ,

        e.name AS [订阅服务器名] ,

        d.subscriber_db AS [订阅数据库]

FROM    distribution.sys.servers a WITH ( NOLOCK )

        INNER JOIN distribution.dbo.MSpublications b WITH ( NOLOCK ) 

ON a.server_id = b.publisher_id

        INNER JOIN distribution.dbo.MSarticles c WITH ( NOLOCK ) 

ON b.publication_id = c.publication_id

        INNER JOIN distribution.dbo.MSsubscriptions d WITH ( NOLOCK ) 

ON b.publication_id = d.publication_id

        AND c.article_id = d.article_id

        INNER JOIN distribution.sys.servers e WITH ( NOLOCK ) 

ON e.server_id = subscriber_id
