--在所有数据库上查找表

IF(OBJECT_ID('tempdb..#TempTables') IS NOT NULL)

BEGIN

DROP TABLE #TempTables

END

GO

SELECT DB_NAME() AS DBName, *

INTO #TempTables

FROM sys.tables

WHERE 1<>1

 

EXEC sp_MSforeachdb N'

USE [?];

INSERT INTO #TempTables

SELECT DB_NAME() AS DBName, *

FROM sys.tables'

 

SELECT * FROM #TempTables T

WHERE T.name LIKE '%prop_ware%'
