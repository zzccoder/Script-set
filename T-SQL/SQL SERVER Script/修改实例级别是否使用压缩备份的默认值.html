--=========================================

--修改实例级别是否使用压缩备份的默认值

USE master;

GO

EXEC sp_configure 'backup compression default', '1';

RECONFIGURE WITH OVERRIDE;


--====================================================

--检查是否默认启用压缩备份

SELECT

CASE WHEN S.value_in_use=1 THEN '默认设置为压缩备份'

ELSE '默认未设置为压缩备份' END 'IsCompressionByDefault'

FROM sys.configurations S

WHERE S.name='backup compression default'
