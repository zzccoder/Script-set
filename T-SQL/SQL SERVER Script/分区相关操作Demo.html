--为新创建的分区置顶文件组

ALTER PARTITION SCHEME [ps_Test1]

NEXT USED [PRIMARY]

GO

--=================================================

--为新分区指定分区边界

ALTER PARTITION FUNCTION [pf_Test1]()

SPLIT RANGE(1500)

GO

--================================================

--合并指定边界的分区

ALTER PARTITION FUNCTION [pf_Test1]()

MERGE RANGE(1500)

--==========================================================

--查看数据所在的分区

SELECT $PARTITION.[pf_Test1](ID) AS PartitionID, * from dbo.TB2
