--======================================================

--查看日志传送主库的备份情况

SELECT db.primary_database

,db.last_backup_date

,db.last_backup_file

FROM [msdb].[dbo].[log_shipping_primary_databases] db

 

--======================================================

--查看日志传送从库的日志拷贝情况

SELECT [primary_server]

,[primary_database]

,[last_copied_file]

,[last_copied_date]

FROM [msdb].[dbo].[log_shipping_secondary]

--======================================================

--查看日志传送从库的还原情况

SELECT db.secondary_database

,db.last_restored_file

,db.last_restored_date

FROM [msdb].[dbo].[log_shipping_secondary_databases] db
