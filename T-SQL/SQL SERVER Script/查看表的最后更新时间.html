--查看表的最后更新时间

--如果索引使用没有记录，则将最后一次索引重建时间记为最后更新时间

SELECT T.name,ISNULL(IU.last_user_update,T.modify_date) last_update_time 

FROM SYS.tables AS T

INNER JOIN SYS.indexes IX

ON IX.object_id=T.object_id

LEFT JOIN SYS.dm_db_index_usage_stats IU

ON IU.index_id=IX.index_id

AND IU.database_id=DB_ID()

AND IU.object_id=IX.object_id

WHERE T.is_ms_shipped=0

AND IX.index_id IN(0,1)
