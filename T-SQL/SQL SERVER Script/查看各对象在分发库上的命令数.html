--查看各对象在分发库上的命令数

USE distribution

GO

SELECT  msa.publisher_db [发布数据库名称] ,

        msa.article [发布表[ ,

        COUNT(1) [命令行数]

FROM    MSarticles (NOLOCK) msa

        JOIN MSrepl_commands (NOLOCK) msc 

ON msa.article_id = msc.article_id

        JOIN MSrepl_transactions (NOLOCK) mst 

ON mst.xact_seqno = msc.xact_seqno

WHERE   publisher_db = 'dbname'

        AND mst.entry_time > GETDATE() - 1

GROUP BY msa.article ,

        msa.publisher_db

ORDER BY COUNT(1) DESC
